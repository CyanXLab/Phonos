   645вҶ’   645вҶ’function tokenizeIPA(word) {
   646вҶ’   646вҶ’    const tokens = [];
   647вҶ’   647вҶ’    let i = 0;
   648вҶ’   648вҶ’    while (i < word.length) {
   649вҶ’   649вҶ’        const ch = word[i];

  1548вҶ’  1548вҶ’function encodeWAV(chunks, sampleRate) {
  1549вҶ’  1549вҶ’    const totalLen = chunks.reduce((s, c) => s + c.length, 0);
  1550вҶ’  1550вҶ’    const buffer = new ArrayBuffer(44 + totalLen * 2);
  1551вҶ’  1551вҶ’    const view = new DataView(buffer);
  1552вҶ’  1552вҶ’
  1553вҶ’  1553вҶ’    const writeStr = (offset, str) => { for (let i = 0; i < str.length; i++) view.setUint8(offset + i, str.charCodeAt(i)); };
  1554вҶ’  1554вҶ’    writeStr(0, 'RIFF');
  1555вҶ’  1555вҶ’    view.setUint32(4, 36 + totalLen * 2, true);
  1556вҶ’  1556вҶ’    writeStr(8, 'WAVE');
  1557вҶ’  1557вҶ’    writeStr(12, 'fmt ');
  1558вҶ’  1558вҶ’    view.setUint32(16, 16, true);
  1559вҶ’  1559вҶ’    view.setUint16(20, 1, true);
  1560вҶ’  1560вҶ’    view.setUint16(22, 1, true);
  1561вҶ’  1561вҶ’    view.setUint32(24, sampleRate, true);
  1562вҶ’  1562вҶ’    view.setUint32(28, sampleRate * 2, true);
  1563вҶ’  1563вҶ’    view.setUint16(32, 2, true);
  1564вҶ’  1564вҶ’    view.setUint16(34, 16, true);
  1565вҶ’  1565вҶ’    writeStr(36, 'data');
  1566вҶ’  1566вҶ’    view.setUint32(40, totalLen * 2, true);
  1567вҶ’  1567вҶ’
  1568вҶ’  1568вҶ’    let offset = 44;
  1569вҶ’  1569вҶ’    for (const chunk of chunks) {
  1570вҶ’  1570вҶ’        for (let i = 0; i < chunk.length; i++) {
  1571вҶ’  1571вҶ’            let s = Math.max(-1, Math.min(1, chunk[i]));
  1572вҶ’  1572вҶ’            s = s < 0 ? s * 0x8000 : s * 0x7FFF;
  1573вҶ’  1573вҶ’            view.setInt16(offset, s, true);
  1574вҶ’  1574вҶ’            offset += 2;
  1575вҶ’  1575вҶ’        }
  1576вҶ’  1576вҶ’    }
  1577вҶ’  1577вҶ’
  1578вҶ’  1578вҶ’    return new Blob([buffer], { type: 'audio/wav' });
  1579вҶ’  1579вҶ’}
